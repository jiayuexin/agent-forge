## 1.1 IAgent 统一接口

整个框架的核心契约，所有生成的 Agent 都必须实现此接口。

```typescript
interface IAgent<TConfig extends AgentConfig = AgentConfig> {
  readonly id: string;
  readonly name: string;
  readonly role: string;
  readonly version: string;
  readonly capabilities: AgentCapability[];
  readonly status: AgentStatus;

  init(config: TConfig): Promise<void>;
  execute(task: AgentTask): Promise<AgentResult>;
  stream(task: AgentTask): AsyncIterable<AgentStreamChunk>;
  destroy(): Promise<void>;
  use(plugin: IPlugin): this;
  on(event: AgentEvent, handler: EventHandler): this;
  off(event: AgentEvent, handler: EventHandler): this;
}

/** Agent 事件类型 — 字符串字面量联合,便于 IDE 自动补全 */
type AgentEvent =
  | 'agent:init'         // 初始化完成
  | 'agent:ready'        // 就绪,等待任务
  | 'agent:execute:start' // 开始执行任务
  | 'agent:execute:end'   // 任务执行完成
  | 'agent:tool:call'     // 工具被调用前
  | 'agent:tool:result'   // 工具返回结果
  | 'agent:llm:chunk'     // LLM 流式输出一个 chunk
  | 'agent:llm:error'     // LLM 调用失败
  | 'agent:error'         // Agent 内部错误
  | 'agent:pause'         // 暂停
  | 'agent:resume'        // 恢复
  | 'agent:destroy';      // 销毁前

/** 事件处理器签名 */
type EventHandler<T = unknown> = (payload: T) => void | Promise<void>;

/** Agent 流式输出 chunk — Agent 层聚合,IProvider 层 chunk 不会直接暴露 */
interface AgentStreamChunk {
  type: 'text' | 'tool_call' | 'tool_result' | 'thinking' | 'done' | 'error';
  content?: string;
  toolCall?: {
    name: string;
    args: Record<string, unknown>;
  };
  toolResult?: {
    name: string;
    output: unknown;
  };
  error?: { code: string; message: string };
  /** chunk 序号,从 0 开始;done 类型时不递增 */
  index: number;
  /** 当前累计 token 数(输入+输出) */
  tokensUsed?: { input: number; output: number };
}

enum AgentStatus {
  UNINITIALIZED = 'uninitialized',
  INITIALIZING = 'initializing',
  READY = 'ready',
  RUNNING = 'running',
  PAUSED = 'paused',
  ERROR = 'error',
  DESTROYED = 'destroyed',
}

interface AgentCapability {
  name: string;
  description: string;
  inputSchema?: JSONSchema;
  outputSchema?: JSONSchema;
}
```

## 1.4 模型注册表（ModelRegistry）

> 编排器核心能力：集中管理多个模型端点，Agent 只需引用模型名即可。

### 设计思路

实际业务中，一个项目可能同时使用多个模型服务：

```
OpenAI 官方 API    → gpt-4o, gpt-4o-mini, o1
私有代理           → gpt-4o（通过代理访问）
Azure OpenAI      → gpt-4o（企业版）
本地 Ollama       → qwen2.5:14b, deepseek-r1:14b
DeepSeek API      → deepseek-chat, deepseek-reasoner
```

同一个模型名（如 `gpt-4o`）可能存在于多个端点。**模型注册表**让编排器统一管理这些端点，Agent 只需要写模型名，编排器自动路由。

### 类型定义

```typescript
/** 单个模型端点 — 一个 baseUrl 下可注册多个模型 */
interface ModelEndpoint {
  /** 端点标识符，用于 agent 引用 */
  id: string;
  /** 端点地址 */
  baseUrl: string;
  /** Provider 类型 */
  provider: 'openai' | 'anthropic' | 'ollama' | string;
  /** API Key（openai/anthropic 需要） */
  apiKey?: string;
  /** 额外参数（anthropicVersion, organization 等） */
  extra?: Record<string, any>;
  /** 该端点下注册的模型列表 */
  models: string[];
}

/** 模型注册表 — 框架初始化时一次性配置 */
interface ModelRegistry {
  /** 所有端点 */
  endpoints: ModelEndpoint[];
  /** 默认端点 id（Agent 不指定模型时的回退） */
  defaultEndpoint?: string;
  /** 默认模型名 */
  defaultModel?: string;
}

/** Agent 引用模型的方式 — 只需模型名 + 可选端点 id */
interface ModelRef {
  /** 模型名（如 'gpt-4o', 'qwen2.5:14b'） */
  model: string;
  /** 指定使用哪个端点（可选，不指定时自动匹配第一个注册了该模型的端点） */
  endpoint?: string;
}
```

### 使用方式

**框架初始化 — 注册所有模型端点：**

```typescript
const framework = new AgentFramework({
  modelRegistry: {
    endpoints: [
      {
        id: 'openai-official',
        baseUrl: 'https://api.openai.com/v1',
        provider: 'openai',
        apiKey: process.env.OPENAI_API_KEY,
        models: ['gpt-4o', 'gpt-4o-mini', 'o1'],
      },
      {
        id: 'openai-proxy',
        baseUrl: 'https://my-proxy.example.com/v1',
        provider: 'openai',
        apiKey: process.env.PROXY_API_KEY,
        models: ['gpt-4o'],
      },
      {
        id: 'ollama-local',
        baseUrl: 'http://localhost:11434',
        provider: 'ollama',
        models: ['qwen2.5:14b', 'deepseek-r1:14b'],
      },
      {
        id: 'deepseek-api',
        baseUrl: 'https://api.deepseek.com/v1',
        provider: 'openai',   // DeepSeek 兼容 OpenAI 协议
        apiKey: process.env.DEEPSEEK_API_KEY,
        models: ['deepseek-chat', 'deepseek-reasoner'],
      },
    ],
    defaultEndpoint: 'openai-official',
    defaultModel: 'gpt-4o',
  },
});
```

**Pipeline 编排 — Agent 只写模型名：**

```typescript
await framework
  .pipeline('智能客服')
  .config({
    defaultModel: 'gpt-4o',          // 流水线级默认
  })
  .add('intent', {
    task: '判断用户意图',
    model: 'gpt-4o-mini',            // 自动路由到 openai-official
  })
  .add('analyze', {
    task: '深度分析问题',
    model: 'deepseek-reasoner',      // 自动路由到 deepseek-api
  })
  .add('draft', {
    task: '起草回复',
    model: {                         // 指定端点
      model: 'gpt-4o',
      endpoint: 'openai-proxy',      // 走代理
    },
  })
  .run();
```

### 模型解析规则

编排器解析 Agent 的模型引用时，按以下规则：

```
1. Agent 指定了 endpoint → 直接用该端点 + 模型名
2. Agent 只指定了 model 名 → 在注册表中查找第一个注册了该模型的端点
3. Agent 都没指定 → 用 Pipeline 级 defaultModel，路由到 defaultEndpoint
4. 找不到匹配的端点 → 报错：ModelNotFoundError
```

### Agent 单独运行

Agent 不在 Pipeline 中单独跑时，可以简单配置一个端点：

```typescript
// 最简写法 — 一个 baseUrl + 一个模型
await agent.init({
  model: {
    provider: 'ollama',
    modelName: 'qwen2.5:14b',
    baseUrl: 'http://localhost:11434',
  },
  systemPrompt: '...',
});
```

> 单独运行时直接用 `ModelConfig`（1.2 节），不走注册表。注册表是编排器的特性。

## 1.8 Pipeline 编排类型

> 参见 [04-集成与编排.md](./04-集成与编排.md) 第 5.3 节的完整设计说明。

```typescript
/** Pipeline 控制信号 — Agent 或拦截器可通过此信号控制流水线流向 */
interface PipelineControlSignal {
  action: 'continue'           // 正常继续下一步
        | 'back'                // 回退到指定步骤重做
        | 'jump'                // 跳转到指定步骤（前进或后退）
        | 'pause'               // 暂停流水线，等待外部输入
        | 'stop'                // 终止流水线
        | 'fork'                // 从当前步骤分叉出并行分支
        | 'replace';            // 用另一个 Agent 替换当前步骤的后续步骤

  targetStep?: string;         // back/jump/replace 的目标步骤名
  message?: string;            // 传给目标步骤的反馈信息
  context?: Record<string, any>; // 附带的上下文数据
  maxRetries?: number;         // back 模式的最大重试次数（默认 1）
  reason?: string;             // 回退/跳转的原因（用于日志和审计）
}

/** 步骤快照 — 每步执行完毕后自动保存 */
interface StepSnapshot {
  stepName: string;
  stepIndex: number;
  timestamp: number;
  input: unknown;
  output: AgentResult;
  duration: number;
  control?: PipelineControlSignal;
  agentState?: unknown;        // Agent 内部状态快照（需实现 ISnapshotable）
}

/** 回退事件记录 */
interface BacktrackEvent {
  fromStep: string;
  toStep: string;
  reason: string;
  retryCount: number;
  maxRetries: number;
  timestamp: number;
  control: PipelineControlSignal;
}

/** Pipeline 执行结果 */
interface PipelineResult {
  finalOutput: AgentResult;
  success: boolean;
  steps: StepSnapshot[];
  backtrackHistory: BacktrackEvent[];
  totalDuration: number;
}

/** Pipeline 全局配置 */
interface PipelineConfig {
  defaultModel?: string;            // Pipeline 级默认模型名（引用 modelRegistry 中的模型）
  maxBacktracks?: number;         // 全局最大回退次数（默认 5），防止无限循环
  defaultMaxRetry?: number;       // 每步默认最大重试次数（默认 2）
  snapshotEnabled?: boolean;      // 是否保存快照（默认 true）
  backtrackLogging?: 'silent' | 'summary' | 'verbose'; // 回退日志级别
  onBacktrack?: (event: BacktrackEvent) => void; // 回退回调
}

/** Agent 快照接口 — Agent 可选实现，支持回退时恢复内部状态 */
interface ISnapshotable {
  snapshot(): Promise<unknown>;          // 保存当前状态
  restore(state: unknown): Promise<void>; // 恢复到指定状态
}
```

## 1.10 IProvider 接口与聊天类型

> Provider 适配层的核心契约。所有 LLM Provider(OpenAI/Anthropic/Ollama/自定义)都必须实现此接口。

```typescript
/** Provider 核心接口 */
interface IProvider {
  /** Provider 标识 */
  readonly provider: string;
  /** 同步聊天 — 返回完整响应 */
  chat(params: ChatParams): Promise<ChatResponse>;
  /** 流式聊天 — 返回 AsyncIterable chunk */
  chatStream(params: ChatParams): AsyncIterable<ChatChunk>;
  /** 验证配置是否合法(如 API Key 格式、baseUrl 可达) */
  validate(): Promise<boolean>;
}

/** 聊天请求参数 */
interface ChatParams {
  /** 消息列表,含 system/user/assistant/tool 角色 */
  messages: Message[];
  /** 可用的工具定义(Function Calling) */
  tools?: ToolDefinition[];
  /** 生成参数 */
  temperature?: number;
  maxTokens?: number;
  stop?: string[];
  /** 请求级超时(ms) */
  timeout?: number;
  /** 请求级追踪 ID,透传给 Provider */
  traceId?: string;
}

/** 聊天响应 — 同步模式 */
interface ChatResponse {
  /** 消息内容(纯文本) */
  content: string;
  /** 结构化输出(若 Agent 配置了结构化输出) */
  structured?: Record<string, unknown>;
  /** 工具调用请求(若 LLM 决定调用工具) */
  toolCalls?: ToolCallRequest[];
  /** token 用量 */
  usage: { input: number; output: number; total: number };
  /** 使用的模型名 */
  model: string;
  /** 结束原因: stop / tool_call / max_tokens / error */
  finishReason: string;
}

/** 聊天流式 chunk */
interface ChatChunk {
  type: 'text' | 'tool_call' | 'tool_result' | 'thinking' | 'done' | 'error';
  /** 文本内容增量 */
  delta?: string;
  /** 工具调用增量 */
  toolCallDelta?: {
    index: number;
    name?: string;
    argsDelta?: string;
  };
  /** 完成的工具调用(仅 type=tool_call 且调用完成时) */
  toolCall?: ToolCallRequest;
  /** 完成的工具结果(仅 type=tool_result) */
  toolResult?: { name: string; output: unknown };
  /** 用量(仅 type=done) */
  usage?: { input: number; output: number; total: number };
  /** 结束原因(仅 type=done) */
  finishReason?: string;
  /** 错误信息(仅 type=error) */
  error?: { code: string; message: string };
}

/** 工具调用请求 — LLM 返回的"我要调用这个工具" */
interface ToolCallRequest {
  /** 工具名(kebab-case) */
  name: string;
  /** 调用参数 */
  args: Record<string, unknown>;
  /** 调用 ID(用于匹配结果) */
  callId: string;
}
```

### Provider Tool Call 格式适配

各 Provider 的 Function Calling 格式不同,IProvider 实现需负责适配:

| Provider | 工具定义格式 | 工具调用格式 | 适配层位置 |
|---|---|---|---|
| OpenAI | `tools[].function` (name/description/parameters) | `tool_calls[].function` (name/arguments) | `OpenAIProvider.ts` 内部转换 |
| Anthropic | `tools[]` (name/description/input_schema) | `content[].type=tool_use` (name/input) | `AnthropicProvider.ts` 内部转换 |
| Ollama | 兼容 OpenAI 格式 | 兼容 OpenAI 格式 | 直接透传 |

框架内部统一使用 `ToolDefinition`(kebab-case name + JSONSchema parameters)和 `ToolCallRequest`(name + args + callId),各 Provider 适配层负责 bidirectional mapping。
